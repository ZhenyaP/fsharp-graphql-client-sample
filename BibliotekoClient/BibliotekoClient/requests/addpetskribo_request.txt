mutation addPetskriboToBiblioteko(
    $bibliotekoId: String!
    $petskribo: InputPetskribo!
    $userId: String!
) 
{
    addPetskribo(
        bibliotekoId: $bibliotekoId
        petskribo: $petskribo
        userId: $userId
    ) 
    {
        id
        registri {
            id
            isbn
            title
            authors
            reviews {
                id
                reviewer
                content {
                    __typename
                    ... on Reaction {
                        reaction
                    }
                    ... on Comment {
                        comment
                    }
                    ... on ReactionAndComment {
                        reaction 
                        comment
                    }
                }
            }
            summary
            imageurl
        }
        logbook
        owner
    }
}
