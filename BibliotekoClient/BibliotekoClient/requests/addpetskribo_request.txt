mutation addPetskriboToBiblioteko(
    $bibliotekoId: String!
    $petskribo: InputPetskribo!
    $uzantoId: String!
) 
{
    addPetskribo(
        bibliotekoId: $bibliotekoId
        petskribo: $petskribo
        uzantoId: $uzantoId
    ) 
    {
        id
        registri {
            id
            isbn
            title
            authors
            recenzos {
                id
                recenzorer
                content {
                    ... on Reaction {
                        reaction
                    }
                    ... on Comment {
                        comment
                    }
                    ... on ReactionAndComment {
                        reaction 
                        comment
                    }
                }
            }
            summary
            imageurl
        }
        logbook
        owner
    }
}
