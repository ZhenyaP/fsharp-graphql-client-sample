mutation removeReaction(
    $reactionId: String!
) 
{
    removeReaction(
        reactionId: $reactionId
    ) 
    {
        id
        isbn
        title
        authors
        reviews {
            id
            reviewer
            content {
                __typename
                ... on Reaction {
                    type
                }
                ... on Comment {
                    type
                }
                ... on ReactionAndComment {
                    reaction {
                        type
                    }
                    comment {
                        type
                    }
                }
            }
        }
        summary
        imageurl
    }
}
