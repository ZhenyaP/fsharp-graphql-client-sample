query getRegistriByIsbn($isbn: String!) {
    registri(isbn: $isbn) {
        id
        isbn
        title
        authors
        recenzos {
            id
            recenzorer
            content {
                ... on Reaction {
                    reaction
                }
                ... on Comment {
                    comment
                }
                ... on ReactionAndComment {
                    reaction 
                    comment
                }
            }
        }
        summary
        imageurl
    }
}

{
  "isbn": "978-1617291326"
}
